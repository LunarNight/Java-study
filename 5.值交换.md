#5.1值交换  

    public class l {
    
        public static void main(String arg[]){
            int a = 5;
            int b = 3;
            changevalue(a,b);
            System.out.println("a="+a+"b="+b);
        }
        public static void changevalue(int a,int b){
            int temp = a;
            a=b;
            b=temp;
            System.out.println("局部函数a="+a+"b="+b);
        }
    }
